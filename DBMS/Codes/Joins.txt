-- INNER JOIN

select course.cid,course.course,hod.cid,hod.Name 
from course
INNER JOIN hod ON course.cid = hod.cid;

-- LEFT JOIN

select course.cid,course.course,hod.cid,hod.Name 
from course
LEFT JOIN hod ON course.cid = hod.cid;

-- RIGHT JOIN

select course.cid,course.course,hod.cid,hod.Name 
from course
RIGHT JOIN hod ON course.cid = hod.cid;

-- FULL OUTER JOIN

select course.cid,course.course,hod.cid,hod.Name 
from course
FULL OUTER JOIN hod ON course.cid = hod.cid;

-- SELF JOIN

select course.cid,course.course,hod.cid,hod.Name 
from course
JOIN hod ON course.cid = hod.cid;
